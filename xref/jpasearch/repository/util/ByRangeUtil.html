19  package jpasearch.repository.util;
20  
21  import static java.lang.Boolean.FALSE;
22  import static java.lang.Boolean.TRUE;
23  
24  import javax.inject.Inject;
25  import javax.inject.Named;
26  import javax.inject.Singleton;
27  import javax.persistence.criteria.CriteriaBuilder;
28  import javax.persistence.criteria.Path;
29  import javax.persistence.criteria.Predicate;
30  import javax.persistence.criteria.Root;
31  
32  import jpasearch.repository.query.selector.Range;
33  
34  /**
35   * Helper to create a predicate out of {@link Range}s.
36   */
37  @SuppressWarnings("unchecked")
38  @Named
39  @Singleton
40  public class ByRangeUtil {
41  
42      @Inject
43      private JpaUtil jpaUtil;
44  
45      public <E> Predicate byRange(Root<E> root, CriteriaBuilder builder, Range<E, ?> range) {
46          if (range.isSet()) {
47              return buildRangePredicate(range, root, builder);
48          } else {
49              return null;
50          }
51      }
52  
53      private <D extends Comparable<? super D>, E> Predicate buildRangePredicate(Range<E, D> range, Root<E> root, CriteriaBuilder builder) {
54          Predicate rangePredicate = null;
55          Path<D> path = jpaUtil.getPath(root, range.getPath());
56          if (range.isBetween()) {
57              rangePredicate = builder.between(path, range.getLowerBound(), range.getHigherBound());
58          } else if (range.isLowerBoundSet()) {
59              if (range.isIncludeLowerBound()) {
60                  rangePredicate = builder.greaterThanOrEqualTo(path, range.getLowerBound());
61              } else {
62                  rangePredicate = builder.greaterThan(path, range.getLowerBound());
63              }
64          } else if (range.isHigherBoundSet()) {
65              if (range.isIncludeHigherBound()) {
66                  rangePredicate = builder.lessThanOrEqualTo(path, range.getHigherBound());
67              } else {
68                  rangePredicate = builder.lessThan(path, range.getHigherBound());
69              }
70          }
71  
72          if (rangePredicate != null) {
73              if (!range.isIncludeNullSet() || (range.getIncludeNull() == FALSE)) {
74                  return rangePredicate;
75              } else {
76                  return builder.or(rangePredicate, builder.isNull(path));
77              }
78          } else {
79              // no from/to is set, but include null or not could be:
80              if (TRUE == range.getIncludeNull()) {
81                  return builder.isNull(path);
82              } else if (FALSE == range.getIncludeNull()) {
83                  return builder.isNotNull(path);
84              }
85          }
86          return null;
87      }
88  }
